extends Control

@onready var GridStuff = $GridContainer
@onready var TurnColour = $TurnColour

var ColourOfPieces = ["WHITE","BLACK"]
var currenty_selected_piece = null

func PlacePiece(new_colorrect,PieceType,PieceColour):
	var new_label = Label.new()
	
	new_label.text = PieceType
	new_colorrect.add_child(new_label)
	new_label.mouse_filter=Control.MOUSE_FILTER_STOP
	new_label.mouse_entered.connect(_on_label_mouse_entered.bind(new_label))
	new_label.mouse_exited.connect(_on_label_mouse_exited.bind(new_label))
	new_label.gui_input.connect(_on_label_gui_input.bind(new_label))
	
	var chosen_color = Color.from_string(ColourOfPieces[PieceColour], Color.WHITE)
	new_label.add_theme_color_override("font_color", chosen_color)
	new_label.set_meta("original_color", chosen_color)
	pass
	
func _on_label_gui_input(event):
	if event is InputEventMouseButton:
		pass
pass	
	
func _on_label_mouse_entered(TargetPiece):
	TargetPiece.add_theme_color_override("font_color", Color.YELLOW)
	pass
	
func _on_label_mouse_exited(TargetPiece):
	var old_color = TargetPiece.get_meta("original_color")
	TargetPiece.add_theme_color_override("font_color", old_color)
	pass
# Called when the node enters the scene tree for the first time.
func _ready() -> void:
	GridStuff.columns = 8
	GridStuff.custom_minimum_size = Vector2(40,40)
	for i in GridStuff.columns:
		for t in range(8):
			var new_colorrect = ColorRect.new()
			if (i+t) % 2 == 0:
				new_colorrect.color=Color(0.803, 0.81, 0.818, 1.0)
			else: 
				new_colorrect.color=Color(0.63, 0.773, 0.456, 1.0)
			new_colorrect.custom_minimum_size=Vector2(40,40)
			if (i == 0 and t == 0) or (i == 0 and t == 7):
				PlacePiece(new_colorrect,"R",1)
			if (i == 7 and t == 0) or (i == 7 and t == 7):
				PlacePiece(new_colorrect,"R",0)
			if (i == 0 and t == 1) or (i == 0 and t == 6):
				PlacePiece(new_colorrect,"H",1)
			if (i == 7 and t == 1) or (i == 7 and t == 6):
				PlacePiece(new_colorrect,"H",0)
			if (i == 0 and t == 2) or (i == 0 and t == 5):
				PlacePiece(new_colorrect,"B",1)
			if (i == 7 and t == 2) or (i == 7 and t == 5):
				PlacePiece(new_colorrect,"B",0)
			if (i == 0 and t == 3):
				PlacePiece(new_colorrect,"Q",1)
			if (i == 7 and t == 3):
				PlacePiece(new_colorrect,"Q",0)
			if (i == 0 and t == 4):
				PlacePiece(new_colorrect,"K",1)
			if (i == 7 and t == 4):
				PlacePiece(new_colorrect,"K",0)
			if i == 1:
				PlacePiece(new_colorrect,"P",1)
			if i == 6:
				PlacePiece(new_colorrect,"P",0)
			GridStuff.add_child(new_colorrect)
	pass
