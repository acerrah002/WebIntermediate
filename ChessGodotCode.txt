extends GridContainer

@onready var Turnrect = $"../TurnRect"

var board_data = []
var Turns = 0 

func _ready():
	columns = 8
	update_turn_visual()
	
	for y in range(8):
		board_data.append([])
		for x in range(8):
			var unit_type = get_starting_unit(x, y) 
			board_data[y].append(unit_type)
			
			var cell = ColorRect.new()
			cell.custom_minimum_size = Vector2(60, 60)
			cell.color = Color.POWDER_BLUE if (x + y) % 2 == 0 else Color.LIGHT_GREEN
			
			# Enable mouse detection
			cell.mouse_filter = Control.MOUSE_FILTER_STOP
			# Pass the 'cell' itself into the function so we can look at its label
			cell.gui_input.connect(_on_cell_input.bind(cell))
			
			add_child(cell)
			
			var label = Label.new()
			label.text = unit_type
			label.horizontal_alignment = HORIZONTAL_ALIGNMENT_CENTER
			label.vertical_alignment = VERTICAL_ALIGNMENT_CENTER
			label.set_anchors_and_offsets_preset(Control.PRESET_FULL_RECT)
			
			if y >= 6:
				label.add_theme_color_override("font_color", Color.WHITE)
			else:
				label.add_theme_color_override("font_color", Color.BLACK)
				
			cell.add_child(label)

func _on_cell_input(event: InputEvent, cell: ColorRect):
	if event is InputEventMouseButton and event.pressed and event.button_index == MOUSE_BUTTON_LEFT:
		# Access the label (it's the only child of the cell)
		var label = cell.get_child(0)
		var text_color = label.get_theme_color("font_color")
		
		print("Text color of this tile is: ", text_color)
		
		# If you want to check specifically for White or Black:
		if text_color == Color.WHITE:
			print("This is a White piece.")
		elif text_color == Color.BLACK:
			print("This is a Black piece.")

func end_turn():
	Turns += 1
	update_turn_visual()

func update_turn_visual():
	Turnrect.color = Color.WHITE if Turns % 2 == 0 else Color.BLACK

func get_starting_unit(x, y):
	if y == 0:
		var units = ["R", "N", "B", "Q", "K", "B", "N", "R"]
		return units[x]
	if y == 1:
		return "P"
	if y == 6:
		return "P"
	if y == 7:
		var units = ["R", "N", "B", "Q", "K", "B", "N", "R"]
		return units[x]
	return ""
